CREATE TABLE R1(Lic_no INT PRIMARY KEY,Fact_no INT,Base_year INT,Contact_no BIGINT NOT NULL);

CREATE TABLE R2(Indus_id BIGINT PRIMARY KEY,percent_of_consumption INT NOT NULL,min_threshold DECIMAL NOT NULL);

CREATE TABLE R3(Sector_id INT PRIMARY KEY,Trade_code INT NOT NULL,Private varchar(3) NOT NULL,Government varchar(3) NOT NULL);

CREATE TABLE R4(Src_id INT PRIMARY KEY,Quantity DECIMAL NOT NULL);

CREATE TABLE R5(Fertilizer_id INT PRIMARY KEY,Composition DECIMAL NOT NULL,Exp_year DATE);

CREATE TABLE R6(CONSTRAINT fk_Lic_No FOREIGN KEY(Lic_No)REFERENCES R1(Lic_No),CONSTRAINT fk_Indus_Id FOREIGN KEY(Indus_Id)REFERENCES R2(Indus_Id));
